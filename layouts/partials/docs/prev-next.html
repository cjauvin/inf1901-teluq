{{/* Ceci est un override que j'ai ajouté pour faire en sorte que la page d'accueil ait un lien "next page", et qu'il soit à droite, même s'il y a aucun lien "prev" */}} {{ $bookSection := default "docs" .Site.Params.BookSection }} {{ if eq $bookSection "*" }} {{ $bookSection = "/" }}{{/* Backward compatibility */}} {{ end }} {{ with .Site.GetPage $bookSection }} {{ $pages := partialCached "docs/prev-next-cache" .Pages }} {{ $homepage := .Site.Home }} {{ $allPages := slice $homepage | append $pages }} {{ $prev := $allPages.Next $ }} {{ $next := $allPages.Prev $ }} {{- if or $prev $next }}
{{ with $prev }} {{ partial "docs/title" . }} {{ end }} {{ with $next }} {{ partial "docs/title" . }} {{ end }}
{{ end -}} {{ end }}
